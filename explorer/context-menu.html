Simple context menu 


                            
                                <script>
                                    function openCtxMenu1() {
                                        win.contextMenu([
                                            {
                                                onclick: () => {
                                                    alert("hello");
                                                },
                                                text: "Show alert",
                                                img: "/etc/icons/star.svg"
                                            },
                                            {
                                                enabled: false,
                                                onclick: () => {
                                                    alert("hello");
                                                },
                                                text: "Disabled item"
                                            }
                                        ]);
                                    }
                                </script>
                                <button onclick="openCtxMenu1()" class="win-button">
                                    Open context menu
                                </button>
